# 快速集成

说明

需要同时引入element-plus

import elementplusMagic from 'elementplus-magic'
import ElementPlus from 'element-plus'
import 'elementplus-magic/dist/elementplus-magic.css'
import 'element-plus/dist/index.css'
app.use(ElementPlus)
app.use(elementplusMagic)
